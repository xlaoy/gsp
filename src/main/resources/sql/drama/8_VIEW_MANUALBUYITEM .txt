CREATE OR REPLACE VIEW VIEW_MANUALBUYITEM AS
SELECT
    RECORDID, ORDERID,DPID, AREAID,QUANTITY,UNITPRICE,ADDTIME,
    PLAYTIME,COSTPRICE,ORIPRICE,TOTALCOST,TOTALFEE,DISCOUNT,
    PLACEID,ITEMID,SETTLEID,SETTLETYPE,DISID,DISFEE,DISCOUNTING,
    COMMISSION,REALCOST,tipid

    WEBDATA.MANUALBUYITEM;
